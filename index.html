Is your Travis test like this?

Enter Docker

.travis.yml

services:
  - docker
install:
  - docker pull jjmerelo/test-perl6
  - docker run -t --entrypoint="/bin/sh" jjmerelo/test-perl6 -c \
      "apk update && apk upgrade && apk add openssl-dev"
script:
  - docker run -t -v  $TRAVIS_BUILD_DIR:/test jjmerelo/test-perl6

That's fast!
